     ```fsharp
     /// Cache for type-check results by absolute file path (single process lifetime)
     /// Stores file's OWN exports only (not merged with caller), so diamond deps are safe.
     let private tcCache = System.Collections.Generic.Dictionary<string, TypeEnv * ConstructorEnv * RecordEnv * Map<string, ModuleExports>>()

     /// Cache for eval results by absolute file path (single process lifetime)
     /// Stores file's OWN exports only (not merged with caller), so diamond deps are safe.
     let private evalCache = System.Collections.Generic.Dictionary<string, Env * Map<string, ModuleValueEnv>>()
     ```

  2. In `loadAndTypeCheckFileImpl`, add cache lookup AFTER the cycle check (fileLoadingStack) but BEFORE file reading.
     CRITICAL: The cache check MUST be placed AFTER the cycle detection check (`fileLoadingStack.Contains`).
     This matches the existing TC pattern where cycle detection takes priority over any shortcut.

     Also CRITICAL: Cache the file's OWN exports, NOT the merged result. This avoids diamond dependency issues where different callers pass different base environments.

     ```fsharp
     let rec loadAndTypeCheckFileImpl resolvedPath cEnv rEnv typeEnv mods =
         // 1. Cycle detection first (MUST happen before cache check)
         if fileLoadingStack.Contains resolvedPath then
             raise (TypeException { ... })
         // 2. Cache check (after cycle detection)
         match tcCache.TryGetValue(resolvedPath) with
         | true, (fileCEnv, fileREnv, fileMods, fileTypeEnv) ->
             // Return caller's env merged with cached file exports
             let mergedCEnv = Map.fold (fun acc k v -> Map.add k v acc) cEnv fileCEnv
             let mergedREnv = Map.fold (fun acc k v -> Map.add k v acc) rEnv fileREnv
             let mergedTypeEnv = Map.fold (fun acc k v -> Map.add k v acc) typeEnv fileTypeEnv
             let mergedMods = Map.fold (fun acc k v -> Map.add k v acc) mods fileMods
             (mergedTypeEnv, mergedCEnv, mergedREnv, mergedMods)
         | false, _ ->
             // 3. Existing logic: file exists check, read, parse, type-check
             if not (File.Exists resolvedPath) then ...
             fileLoadingStack.Add resolvedPath |> ignore
             let prevFile = ...
             try
                 ...
                 // typeCheckModuleWithPrelude returns file's OWN exports
                 // (fileCEnv, fileREnv, fileMods, fileTypeEnv)
                 // Cache the file's own exports BEFORE merging with caller
                 tcCache.[resolvedPath] <- (fileCEnv, fileREnv, fileMods, fileTypeEnv)
                 // Then merge with caller env as before
                 let mergedCEnv = Map.fold (fun acc k v -> Map.add k v acc) cEnv fileCEnv
                 ...
                 (mergedTypeEnv, mergedCEnv, mergedREnv, mergedMods)
             finally
                 ...
     ```

  3. In `loadAndEvalFileImpl`, add cache lookup similarly.

     ```fsharp
     and loadAndEvalFileImpl resolvedPath recEnv modEnv env =
         // Eval skips cycle detection (TC phase catches cycles before eval runs).
         // Cache check is safe at top since cycles are impossible at eval time.
         match evalCache.TryGetValue(resolvedPath) with
         | true, (fileEnv, fileModEnv) ->

             let mergedModEnv = Map.fold (fun acc k v -> Map.add k v acc) modEnv fileModEnv
             (mergedEnv, mergedModEnv)
         | false, _ ->
             if not (File.Exists resolvedPath) then ...
             let prevFile = ...
             try
                 ...
                 let (fileEnv, fileModEnv) = Eval.evalModuleDecls recEnv modEnv env decls
                 // Cache file's own exports BEFORE merging with caller
                 evalCache.[resolvedPath] <- (fileEnv, fileModEnv)
                 // Then merge with caller env as before

                 let mergedModEnv = Map.fold (fun acc k v -> Map.add k v acc) modEnv fileModEnv
                 (mergedEnv, mergedModEnv)
             finally
                 ...
     ```

  4. Do NOT add cache clearing -- caches are process-scoped (single CLI invocation). They are naturally garbage collected when the process exits.

  KEY DESIGN DECISION: Cache file's own exports, not merged results.
  Why: In a diamond pattern (A imports B and C, both import D), B and C may pass different caller environments to D. If we cached the merged result from B's import of D, C would get B's environment leaked in. By caching only D's own exports and re-merging with each caller's environment, each caller gets correct results.
