**In Parser.fsy:**

Add grammar rules for type declarations (after Decl rules):

```fsharp
TypeDeclaration:
    | TYPE IDENT TypeParams EQUALS Constructors TypeDeclContinuation
        { TypeDecl($2, $3, $5, mkSpan parseState 1 6) :: $6 }

TypeParams:
    | (* empty *)  { [] }
    | IDENT        { [$1] }
    | IDENT TypeParams { $1 :: $2 }

Constructors:
    | Constructor  { [$1] }
    | Constructor PIPE Constructors { $1 :: $3 }

Constructor:
    | IDENT  { Constructor($1, None, mkSpan parseState 1 1) }
    | IDENT OF TypeExpr  { Constructor($1, Some $3, mkSpan parseState 1 3) }

TypeDeclContinuation:
    | (* empty *)  { [] }
    | AND TypeDeclaration  { $2 }
```

Extend Decl rule to include TypeDeclaration:
```fsharp
Decl:
    | LET IDENT EQUALS Expr  { LetDecl($2, $4, mkSpan parseState 1 4) }
    | LET IDENT EQUALS INDENT Expr DEDENT  { LetDecl($2, $5, mkSpan parseState 1 6) }
    | LET IDENT ParamList EQUALS Expr  { LetDecl($2, List.foldBack (fun p body -> Lambda(p, body, unknownSpan)) $3 $5, mkSpan parseState 1 5) }
    | LET IDENT ParamList EQUALS INDENT Expr DEDENT  { LetDecl($2, List.foldBack (fun p body -> Lambda(p, body, unknownSpan)) $3 $6, mkSpan parseState 1 7) }
    | TypeDeclaration  { List.map Ast.TypeDecl $1 }
```

Note: TypeDeclaration returns a list (for mutually recursive types), so map each to Decl.TypeDecl.

```fsharp
let parseTypeDecl input =
    let lexbuf = LexBuffer<_>.FromString input
    lexbuf.EndPos <- { lexbuf.EndPos with FileName = "test.fun" }
    try
        Parser.parseModule Lexer.tokenize lexbuf
    with
    | e -> failwithf "Parse error: %s" e.Message
```

Add tests:

```fsharp
[<Fact>]
let testParseSimpleADT() =
    let input = "type Option = None | Some of int"
    let result = parseTypeDecl input
    match result with
    | Module([TypeDecl(TypeDecl("Option", [], [
        Constructor("None", None, _)
        Constructor("Some", Some(TEInt), _)
    ], _))], _) -> ()
    | _ -> failwith "Expected simple ADT with two constructors"

[<Fact>]
let testParseADTWithTypeParam() =
    let input = "type Option 'a = None | Some of 'a"
    let result = parseTypeDecl input
    match result with
    | Module([TypeDecl(TypeDecl("Option", ["'a"], [
        Constructor("None", None, _)
        Constructor("Some", Some(TEVar "'a"), _)
    ], _))], _) -> ()
    | _ -> failwith "Expected ADT with type parameter"

[<Fact>]
let testParseRecursiveADT() =
    let input = "type Tree = Leaf | Node of Tree * int * Tree"
    let result = parseTypeDecl input
    match result with
    | Module([TypeDecl(TypeDecl("Tree", [], [
        Constructor("Leaf", None, _)
        Constructor("Node", Some(TETuple _), _)
    ], _))], _) -> ()
    | _ -> failwith "Expected recursive ADT"

[<Fact>]
let testParseMutuallyRecursiveADT() =
    let input = "type Expr = Lit of int | Arith of ArithExpr\nand ArithExpr = Add of Expr * Expr"
    let result = parseTypeDecl input
    match result with
    | Module([
        TypeDecl(TypeDecl("Expr", _, _, _))
        TypeDecl(TypeDecl("ArithExpr", _, _, _))
    ], _) -> ()
    | _ -> failwith "Expected mutually recursive ADTs"
```
