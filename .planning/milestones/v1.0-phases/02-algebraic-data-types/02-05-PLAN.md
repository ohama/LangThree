Constructor expressions conflict with variables (both use IDENT). Distinguish by uppercase check:

```fsharp
AtomExpr:
    | IntLiteral  { Number($1, mkSpan parseState 1 1) }
    | TRUE  { Bool(true, mkSpan parseState 1 1) }
    | FALSE  { Bool(false, mkSpan parseState 1 1) }
    | STRING  { String($1, mkSpan parseState 1 1) }
    | IDENT  {
        // Uppercase = constructor (nullary), lowercase = variable
        if System.Char.IsUpper($1.[0]) then
            Constructor($1, None, mkSpan parseState 1 1)
        else
            Var($1, mkSpan parseState 1 1)
    }
    | LPAREN Expr RPAREN  { $2 }
    | LPAREN Expr COMMA ExprList RPAREN  { Tuple($2 :: $4, mkSpan parseState 1 5) }
    | LBRACKET RBRACKET  { EmptyList(mkSpan parseState 1 2) }
    | LBRACKET ExprList RBRACKET  { List($2, mkSpan parseState 1 3) }

AppExpr:
    | AtomExpr  { $1 }
    | AppExpr AtomExpr  {
        // Special case: Constructor application
        match $1 with
        | Constructor(name, None, span) ->
            // Convert nullary constructor + arg to constructor with arg
            Constructor(name, Some $2, mkSpan parseState 1 2)
        | _ ->
            App($1, $2, mkSpan parseState 1 2)
    }
```

**Update Match evaluation for ConstructorPat** (existing Match case, around line 100-130):

```fsharp
    | Match (scrut, clauses, _) ->
        let scrutValue = eval env scrut
        // Try each clause
        let rec tryMatch = function
            | [] -> failwith "Non-exhaustive pattern match"  // Will be caught by exhaustiveness checker
            | (pat, body) :: rest ->
                match matchPattern pat scrutValue with
                | Some bindings ->
                    let env' = Map.fold (fun acc k v -> Map.add k v acc) env bindings
                    eval env' body
                | None -> tryMatch rest
        tryMatch clauses

and matchPattern (pat: Pattern) (value: Value) : Map<string, Value> option =
    match (pat, value) with
    | (VarPat (x, _), v) ->
        Some (Map.singleton x v)

    | (WildcardPat _, _) ->
        Some Map.empty

    | (TuplePat (pats, _), TupleValue vs) ->
        if List.length pats = List.length vs then
            let results = List.zip pats vs |> List.map (fun (p, v) -> matchPattern p v)
            if List.forall Option.isSome results then
                let bindings = results |> List.choose id |> List.fold (fun acc m -> Map.fold (fun acc' k v -> Map.add k v acc') acc m) Map.empty
                Some bindings
            else
                None
        else
            None

    | (ConsPat (hPat, tPat, _), ListValue (h :: t)) ->
        match matchPattern hPat h with
        | Some hBindings ->
            match matchPattern tPat (ListValue t) with
            | Some tBindings ->
                let bindings = Map.fold (fun acc k v -> Map.add k v acc) hBindings tBindings
                Some bindings
            | None -> None
        | None -> None

    | (EmptyListPat _, ListValue []) ->
        Some Map.empty

    | (ConstPat (c, _), v) ->
        let matches = match (c, v) with
                      | (IntConst n, IntValue m) -> n = m
                      | (BoolConst b, BoolValue c) -> b = c
                      | _ -> false
        if matches then Some Map.empty else None

    | (ConstructorPat (name, argPatOpt, _), DataValue (ctorName, argValOpt)) ->  // NEW
        if name = ctorName then
            match (argPatOpt, argValOpt) with
            | (None, None) ->
                // Nullary constructor
                Some Map.empty
            | (Some argPat, Some argVal) ->
                // Constructor with argument
                matchPattern argPat argVal
            | _ ->
                // Arity mismatch (caught by type checker)
                None
        else
            // Different constructor
            None

    | _ ->
        // Pattern doesn't match value
        None
```
