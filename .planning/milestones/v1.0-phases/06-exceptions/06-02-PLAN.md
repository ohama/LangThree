```fsharp
    let s1, argTy = synth ctorEnv recEnv ctx env arg

```
Also add Raise to check mode -- delegate to synth (like other expressions that synthesize).

**TryWith (synth mode):**
```fsharp
| TryWith(body, handlers, span) ->
    let s1, bodyTy = synth ctorEnv recEnv ctx env body
    // Type check each handler clause (same pattern as Match)
    let folder (s, _idx) (pat, guard, expr) =
        let patEnv, patTy = inferPattern ctorEnv pat
        let s' = unifyWithContext ctx [] span (apply s patTy) TExn
        let clauseEnv = Map.fold (fun acc k v -> Map.add k v acc)
                            (applyEnv s' (applyEnv s env)) patEnv
        // Type check when guard if present (must be bool)
        let sGuard = match guard with
                     | Some g ->
                         let sg, gTy = synth ctorEnv recEnv ctx clauseEnv g
                         let sg' = unifyWithContext ctx [] span (apply sg gTy) TBool
                         compose sg' sg
                     | None -> Type.empty
        let clauseEnv' = applyEnv sGuard clauseEnv
        let s'', exprTy = synth ctorEnv recEnv ctx clauseEnv' expr
        let sUnify = unifyWithContext ctx [] span (apply s'' (apply sGuard (apply s' (apply s bodyTy)))) (apply s'' exprTy)
        (compose sUnify (compose s'' (compose sGuard (compose s' s))), _idx + 1)
    let finalS, _ = List.fold folder (s1, 0) handlers
    (finalS, apply finalS bodyTy)
```
Also add TryWith to check mode.

**Exhaustive.fs:**
Update exhaustiveness checking to handle TExn as an open type:
- When checking exception handler patterns (pattern type is TExn), skip exhaustiveness checking OR emit a W0003 warning for handlers without a wildcard pattern
- The key change: in the function that determines the constructor set for a type, TExn should return an empty/unknown set (since it's open), making any pattern match against TExn inherently non-exhaustive
- If the existing exhaustiveness code calls a function to get "all constructors of type X", add a case for TExn that returns a sentinel indicating open type (always non-exhaustive)
- Guarded patterns: a pattern with a `when` guard should be treated as potentially non-matching for exhaustiveness purposes. In the Maranget algorithm, treat guarded clauses as if they don't cover any cases (or mark them as partial). The simplest approach: when building the pattern matrix, if a clause has a guard, DON'T include it in the exhaustiveness matrix (it might not match even if the pattern matches).
