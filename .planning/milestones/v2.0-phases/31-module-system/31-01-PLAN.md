    ```fsharp
    open System.IO

    let resolveImportPath (importPath: string) (importingFile: string) : string =
        if Path.IsPathRooted importPath then
            importPath
        else
            let baseDir =
                if not (System.String.IsNullOrEmpty importingFile)
                   && importingFile <> "<unknown>"
                   && importingFile <> "<expr>"
                   && importingFile <> "test"
                   && File.Exists importingFile then
                    Path.GetDirectoryName(Path.GetFullPath importingFile)
                else
                    Directory.GetCurrentDirectory()
            Path.GetFullPath(Path.Combine(baseDir, importPath))
    ```

    (2) `loadAndTypeCheckFile` — add after `resolveImportPath`. Mirrors the Prelude.fs
    `loadSingleFile` pattern (see RESEARCH.md code examples). Takes a resolved absolute
    path, a `Set<string>` visited set (for cycle detection), and the current type envs.
    Returns `(TypeEnv * ConstructorEnv * RecordEnv)` or raises `TypeException`.

    ```fsharp
    let rec loadAndTypeCheckFile
        (resolvedPath: string)
        (visited: Set<string>)
        (cEnv: ConstructorEnv)
        (rEnv: RecordEnv)
        (typeEnv: TypeEnv) : TypeEnv * ConstructorEnv * RecordEnv =
        if Set.contains resolvedPath visited then
            raise (TypeException {
                Kind = CircularModuleDependency [resolvedPath]
                Span = unknownSpan; Term = None; ContextStack = []; Trace = [] })
        let visited' = Set.add resolvedPath visited
        if not (File.Exists resolvedPath) then
            raise (TypeException {
                Kind = UnresolvedModule resolvedPath
                Span = unknownSpan; Term = None; ContextStack = []; Trace = [] })
        let source = File.ReadAllText resolvedPath
        let m = Prelude.parseModuleFromString source resolvedPath
        match typeCheckModuleWithPrelude cEnv rEnv typeEnv m with
        | Ok (_warnings, fileCEnv, fileREnv, _mods, fileTypeEnv) ->
            let mergedCEnv = Map.fold (fun acc k v -> Map.add k v acc) cEnv fileCEnv
            let mergedREnv = Map.fold (fun acc k v -> Map.add k v acc) rEnv fileREnv
            let mergedTypeEnv = Map.fold (fun acc k v -> Map.add k v acc) typeEnv fileTypeEnv
            (mergedTypeEnv, mergedCEnv, mergedREnv)
        | Error diag -> raise (TypeException diag)
    ```

    NOTE: `Prelude.parseModuleFromString` is private in Prelude.fs. Change it to `internal`
    (or add a public wrapper `Prelude.parseModuleFromString`) so TypeCheck.fs can call it.
    Alternatively, duplicate the parse logic inline. Prefer making it `internal` — change
    `let private parseModuleFromString` to `let parseModuleFromString` in Prelude.fs.

    (3) Add `FileImportDecl` match arm in `typeCheckDecls` fold, after the `OpenDecl` arm:

    ```fsharp
        let (env', cEnv', rEnv') = loadAndTypeCheckFile resolvedPath Set.empty cEnv rEnv env
        (env', cEnv', rEnv', mods, warns)
    ```

    NOTE: The `typeCheckDecls` fold uses `env` for the type environment (not a separate

    **Eval.fs** — add two helpers and a new match arm:

    (2) `loadAndEvalFile` — mirrors Prelude.fs eval section. Takes the resolved path,

    ```fsharp
    let loadAndEvalFile
        (resolvedPath: string)
        (recEnv: RecordEnv)
        (modEnv: ModuleEnv)
        (env: Env) : Env * ModuleEnv =
        let source = File.ReadAllText resolvedPath
        // Re-parse (type checking already validated; we just need the AST)
        let m = Prelude.parseModuleFromString source resolvedPath
        let decls =
            match m with
            | Ast.Module(d, _) | Ast.NamedModule(_, d, _) | Ast.NamespacedModule(_, d, _) -> d
            | Ast.EmptyModule _ -> []
        let (fileEnv, fileModEnv) = evalModuleDecls recEnv modEnv env decls
        // Merge: all file bindings shadow/add to current env
        let mergedEnv = Map.fold (fun acc k v -> Map.add k v acc) env fileEnv
        let mergedModEnv = Map.fold (fun acc k v -> Map.add k v acc) modEnv fileModEnv
        (mergedEnv, mergedModEnv)
    ```
