**File 3: gadt-poly-recursive.flt**
Test: recursive GADT evaluator with Add constructor (COV-02).

```
// Test: recursive GADT evaluator — Add (IntLit 10, IntLit 20) = 30
// --- Command: /Users/ohama/vibe-coding/LangThree/src/LangThree/bin/Release/net10.0/LangThree %input
// --- Input:
type Expr 'a = IntLit : int -> int Expr | BoolLit : bool -> bool Expr | Add : int Expr * int Expr -> int Expr

let result =
    let rec eval e = (match e with | IntLit n -> n | BoolLit b -> if b then 1 else 0 | Add (a, b) -> eval a + eval b : int) in
    eval (Add (IntLit 10, IntLit 20))
// --- Output:
30
```
