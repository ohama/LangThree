IMPORTANT: All new bindings go AFTER all existing bindings (after `let (++) ...`) so they can reference `any`, `nth`, `hd`, `tl` which are already defined above.

Add the following in this exact order:

```fsharp
    // Aliases for existing functions
    let head xs = hd xs
    let tail xs = tl xs
    let exists pred xs = any pred xs
    let item n xs = nth n xs
    let isEmpty xs = match xs with | [] -> true | _ -> false

    // List.sort — insertion sort (works for IntValue, StringValue, CharValue, BoolValue)
    let rec _insert x xs =
        match xs with
        | [] -> [x]
        | h :: t -> if x < h then x :: h :: t else h :: _insert x t
    let rec sort xs =
        match xs with
        | [] -> []
        | h :: t -> _insert h (sort t)

    // List.sortBy — wraps list_sort_by builtin (calls user closure for key extraction)
    let sortBy f xs = list_sort_by f xs

    // List.mapi — map with index
    let rec _mapi_helper f i xs =
        match xs with
        | [] -> []
        | h :: t -> f i h :: _mapi_helper f (i + 1) t
    let mapi f xs = _mapi_helper f 0 xs

    // List.tryFind — returns Some/None
    let rec tryFind pred xs =
        match xs with
        | [] -> None
        | h :: t -> if pred h then Some h else tryFind pred t

    // List.choose — map + filter combined (keeps Some results, unwrapped)
    let rec choose f xs =
        match xs with
        | [] -> []
        | h :: t ->
            match f h with
            | Some v -> v :: choose f t
            | None -> choose f t

    // List.distinctBy — deduplicate by key function (O(n^2), correct for small lists)
    let rec _distinctBy_helper f seen xs =
        match xs with
        | [] -> []
        | h :: t ->
            let key = f h
            if any (fun k -> k = key) seen
            then _distinctBy_helper f seen t
            else h :: _distinctBy_helper f (key :: seen) t
    let distinctBy f xs = _distinctBy_helper f [] xs

    // List.ofSeq — convert native collection to immutable list
    let ofSeq coll = list_of_seq coll
```

The leading underscores on helper functions (_insert, _mapi_helper, _distinctBy_helper) follow the convention noted in the research to make helpers visually distinguishable, though they remain accessible.
