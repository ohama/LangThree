```fsharp
/// Merge a module's exports into current environments (for `open` directive)
let openModuleExports (modExports: ModuleExports) (typeEnv: TypeEnv) (ctorEnv: ConstructorEnv) (recEnv: RecordEnv) =
    let typeEnv' = Map.fold (fun acc k v -> Map.add k v acc) typeEnv modExports.TypeEnv
    let ctorEnv' = Map.fold (fun acc k v -> Map.add k v acc) ctorEnv modExports.CtorEnv
    let recEnv' = Map.fold (fun acc k v -> Map.add k v acc) recEnv modExports.RecEnv
    (typeEnv', ctorEnv', recEnv')

/// Resolve a module by name from available modules map
let resolveModule (modules: Map<string, ModuleExports>) (name: string) (span: Span) : ModuleExports =
    match Map.tryFind name modules with
    | Some m -> m
    | None ->
        raise (TypeException {
            Kind = UnresolvedModule name
            Span = span; Term = None; ContextStack = []; Trace = [] })

/// Detect circular dependencies using DFS 3-color algorithm
/// Returns Some(cycle path) if a cycle is found, None otherwise
let detectCircularDeps (graph: Map<string, string list>) : string list option =
    let visited = System.Collections.Generic.HashSet<string>()
    let inStack = System.Collections.Generic.HashSet<string>()
    let rec dfs path node =
        if inStack.Contains(node) then
            let cycle = (path |> List.rev |> List.skipWhile ((<>) node)) @ [node]
            Some cycle
        elif visited.Contains(node) then None
        else
            visited.Add(node) |> ignore
            inStack.Add(node) |> ignore
            let deps = Map.tryFind node graph |> Option.defaultValue []
            let result = deps |> List.tryPick (dfs (node :: path))
            inStack.Remove(node) |> ignore
            result
    graph |> Map.toList |> List.tryPick (fun (node, _) -> dfs [] node)

/// Build dependency graph from open directives in declarations
let buildDependencyGraph (moduleName: string) (decls: Decl list) : Map<string, string list> =
    let rec collect currentModule decls =
        decls |> List.collect (fun d ->
            match d with
            | OpenDecl (path, _) ->
                [(currentModule, String.concat "." path)]
            | ModuleDecl (name, innerDecls, _) ->
                collect name innerDecls
            | _ -> [])
    collect moduleName decls
    |> List.groupBy fst
    |> List.map (fun (k, vs) -> (k, vs |> List.map snd))
    |> Map.ofList
```

This function folds over declarations:
- `LetDecl(name, body, _)` -> synth type, add to TypeEnv, collect match warnings
- `Decl.TypeDecl td` -> elaborate, add to CtorEnv
- `Decl.RecordTypeDecl rd` -> elaborate, add to RecEnv
- `ModuleDecl(name, innerDecls, span)` ->
  - Check for duplicate module name (raise E0503 if name already in modules map)
  - Recursively call `typeCheckDecls` with current envs for the inner decls
  - Build a `ModuleExports` from the inner result
  - Add to modules map under `name`
- `OpenDecl(path, span)` ->
  - For v1, path is a single module name (string list with one element)
  - Look up the module name in the current modules map
  - If not found, raise E0504 ForwardModuleReference (since we process top-to-bottom, not found means either undefined or forward reference)
  - If found, merge its exports into current TypeEnv/CtorEnv/RecEnv using `openModuleExports`
- `NamespaceDecl(path, innerDecls, span)` ->
  - Treat as naming prefix only for v1 (no actual scoping restriction)
  - Process innerDecls normally with typeCheckDecls
  - The namespace path is stored in AST but has no effect on type checking in v1

**3. Update `typeCheckModule` main body:**
