**Test helper pattern** (follow existing test files):
```fsharp
module ModuleTests

open Xunit
open FSharp.Text.Lexing
open Ast
open Eval
open TypeCheck
open Diagnostic

/// Helper: parse, typecheck, and evaluate a module string
let evalModule (input: string) : Value * Diagnostic list =
    let lexbuf = LexBuffer<char>.FromString input
    Lexer.setInitialPos lexbuf "<test>"
    let m = Parser.parseModule Lexer.tokenize lexbuf
    match TypeCheck.typeCheckModule m with
    | Error diag -> failwithf "Type error: %s" (formatDiagnostic diag)
    | Ok (warnings, recEnv, modules) ->
        let decls = match m with
                    | Module(d, _) | NamedModule(_, d, _) | NamespacedModule(_, d, _) -> d
                    | EmptyModule _ -> []
        let initialEnv = Prelude.loadPrelude()
        let finalEnv, moduleEnv = Eval.evalModuleDecls recEnv Map.empty initialEnv decls
        // Evaluate last let binding
        match decls |> List.rev |> List.tryPick (function LetDecl(_, body, _) -> Some body | _ -> None) with
        | Some lastBody -> (Eval.eval recEnv moduleEnv finalEnv lastBody, warnings)
        | None -> (IntValue 0, warnings)

/// Helper: parse and typecheck, expecting a type error
let expectTypeError (input: string) (expectedCode: string) =
    let lexbuf = LexBuffer<char>.FromString input
    Lexer.setInitialPos lexbuf "<test>"
    let m = Parser.parseModule Lexer.tokenize lexbuf
    match TypeCheck.typeCheckModule m with
    | Error diag ->
        match diag.Code with
        | Some code -> Assert.Equal(expectedCode, code)
        | None -> failwith "Expected error code but got None"
    | Ok _ -> failwith "Expected type error but got Ok"
```

**SC5: Circular dependency / ordering enforcement**

NOTE ON SC5 AND E0501: In the v1 single-file model, true circular dependencies between nested modules cannot occur because declarations are processed top-to-bottom. Module A cannot open Module B if B is defined after A, because B doesn't exist yet. This means:
- E0504 (ForwardModuleReference) is the enforcement mechanism that PREVENTS circular dependencies from ever forming
- E0501 (CircularModuleDependency) exists for future multi-file support where files can reference each other
- The `detectCircularDeps` graph algorithm is implemented and ready for multi-file, but unreachable in v1 single-file

Therefore, E0504 serves as the PROXY test for SC5. The forward reference check is the v1 mechanism that ensures module ordering is valid and circular dependencies cannot form.

```fsharp
[<Fact>]
let ``forward module reference produces error - enforces ordering that prevents cycles`` () =
    // SC5 proxy: In single-file v1, forward references are what would create
    // circular dependencies. E0504 prevents them, making E0501 unreachable.
    // This test verifies the ordering enforcement that prevents cycles.
    expectTypeError """
open NotYetDefined

module NotYetDefined =
    let x = 1

let result = x
""" "E0504"

[<Fact>]
let ``forward reference in nested module produces error`` () =
    // Another SC5 proxy: forward reference from inside a nested module
    expectTypeError """
module A =
    open B
    let x = 1

module B =
    let y = 2

let result = 0
""" "E0504"

[<Fact>]
let ``valid top-to-bottom ordering succeeds - no false circular dep error`` () =
    // Verify that valid ordering (A defined before B opens A) works fine
    let result, _ = evalModule """
module A =
    let x = 1

module B =
    open A
    let y = x + 1

let result = B.y
"""
    Assert.Equal(IntValue 2, result)
```
