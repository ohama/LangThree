**1. Modify FieldAccess evaluation** to check for module access first, including constructor resolution:

```fsharp
| FieldAccess (expr, fieldName, span) ->
    // First try: evaluate normally and check if it's a record
    // But also check if the expr is a Var that names a module
    match expr with
    | Var (name, _) when Map.containsKey name moduleEnv ->
        // Module qualified access: Module.member
        let modEnv = Map.find name moduleEnv
        // Try as value first
        match Map.tryFind fieldName modEnv.Values with
        | Some value -> value
        | None ->
            // Try as constructor
            match Map.tryFind fieldName modEnv.CtorEnv with
            | Some ctorValue -> ctorValue
            | None ->
                // Try as nested submodule (for chained access like A.B where B is a submodule)
                // The outer FieldAccess will resolve the final member
                failwithf "Module %s has no member or constructor %s" name fieldName
    | FieldAccess (innerExpr, innerField, _) ->
        // Chained access: check if inner resolves to a module
        // e.g., A.B.c where A.B is a submodule
        match innerExpr with
        | Var (modName, _) when Map.containsKey modName moduleEnv ->
            let outerMod = Map.find modName moduleEnv
            match Map.tryFind innerField outerMod.SubModules with
            | Some innerMod ->
                // Try value then constructor in inner module
                match Map.tryFind fieldName innerMod.Values with
                | Some value -> value
                | None ->
                    match Map.tryFind fieldName innerMod.CtorEnv with
                    | Some ctorValue -> ctorValue
                    | None -> failwithf "Module %s.%s has no member %s" modName innerField fieldName
            | None ->
                // Fall through to record field access
                let v = eval recEnv moduleEnv env expr
                match v with
                | RecordValue (_, fields) ->
                    match Map.tryFind fieldName fields with
                    | Some valueRef -> !valueRef
                    | None -> failwithf "Field not found: %s" fieldName
                | _ -> failwithf "Field access on non-record/module value: %s" (formatValue v)
        | _ ->
            // Regular record field access
            let v = eval recEnv moduleEnv env expr
            match v with
            | RecordValue (_, fields) ->
                match Map.tryFind fieldName fields with
                | Some valueRef -> !valueRef
                | None -> failwithf "Field not found: %s" fieldName
            | _ -> failwithf "Field access on non-record value: %s" (formatValue v)
    | _ ->
        // Regular record field access
        match eval recEnv moduleEnv env expr with
        | RecordValue (_, fields) ->
            match Map.tryFind fieldName fields with
            | Some valueRef -> !valueRef
            | None -> failwithf "Field not found: %s" fieldName
        | v -> failwithf "Field access on non-record value: %s" (formatValue v)
```

**2. Add `evalModuleDecls` function** to process module declarations and build environments, including constructor collection:

```fsharp
/// Evaluate module declarations, building value and module environments
let evalModuleDecls
    (recEnv: RecordEnv)
    (moduleEnv: Map<string, ModuleValueEnv>)
    (initialEnv: Env)
    (decls: Decl list)
    : Env * Map<string, ModuleValueEnv> =
    decls
    |> List.fold (fun (env, modEnv) decl ->
        match decl with
        | LetDecl(name, body, _) ->
            let value = eval recEnv modEnv env body
            (Map.add name value env, modEnv)
        | ModuleDecl(name, innerDecls, _) ->
            let innerEnv, innerModEnv = evalModuleDecls recEnv modEnv env innerDecls
            // Extract only the bindings added by this module (not inherited ones)
            let moduleValues =
                innerEnv
                |> Map.filter (fun k _ -> not (Map.containsKey k env))
            // Collect constructors: any value in moduleValues that is a constructor
            // (ConstructorValue or lambda wrapping a constructor) should also be in CtorEnv.
            // Additionally, scan TypeDecl siblings to find constructor names from the
            // same module scope.
            let ctorNames =
                innerDecls |> List.collect (fun d ->
                    match d with
                    | Decl.TypeDecl (TypeDecl(_, _, ctors, _)) ->
                        ctors |> List.map (fun (CtorDecl(name, _, _)) -> name)
                    | _ -> [])
            let ctorEnv =
                ctorNames
                |> List.choose (fun cname ->
                    Map.tryFind cname innerEnv |> Option.map (fun v -> (cname, v)))
                |> Map.ofList
            let modValEnv = {
                Values = moduleValues
                CtorEnv = ctorEnv
                RecEnv = recEnv  // Simplified: share parent recEnv
                SubModules = innerModEnv
            }
            (env, Map.add name modValEnv modEnv)
        | OpenDecl(path, _) ->
            // For single-segment path, look up directly
            match path with
            | [name] ->
                match Map.tryFind name modEnv with
                | Some modValEnv ->
                    let env' = Map.fold (fun acc k v -> Map.add k v acc) env modValEnv.Values
                    // Also merge constructors into the value env so they're accessible unqualified
                    let env'' = Map.fold (fun acc k v -> Map.add k v acc) env' modValEnv.CtorEnv
                    (env'', modEnv)
                | None -> (env, modEnv)  // Already caught by type checker
            | _ -> (env, modEnv)  // Multi-segment paths: v2
        | _ -> (env, modEnv)  // TypeDecl, RecordTypeDecl handled elsewhere
    ) (initialEnv, moduleEnv)
```

Key difference from previous version: `evalModuleDecls` now:
1. Scans `TypeDecl` siblings inside a `ModuleDecl` to find constructor names
2. Builds a `CtorEnv` mapping constructor names to their runtime values
3. On `open`, merges both `Values` AND `CtorEnv` into the current scope

This ensures `Shapes.Circle 5` works (qualified constructor access) and `open Shapes; Circle 5` works (unqualified after open).
