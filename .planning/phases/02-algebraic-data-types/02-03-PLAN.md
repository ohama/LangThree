```fsharp
let rec inferPattern (ctorEnv: ConstructorEnv) (pat: Pattern) (ty: Type) : TypeEnv * Subst =
    match pat with
    | VarPat (x, _) ->
        // Variable binds to the type
        (Map.singleton x (Scheme([], ty)), empty)

    | WildcardPat _ ->
        // Wildcard matches anything, binds nothing
        (Map.empty, empty)

    | TuplePat (pats, _) ->
        // Match tuple type against pattern list
        let tvs = List.init (List.length pats) (fun _ -> fresh())
        let s1 = unify ty (TTuple tvs)
        let (envs, subs) =
            List.zip pats tvs
            |> List.map (fun (p, t) ->
                let (env, s) = inferPattern ctorEnv p (apply s1 t)
                (env, compose s s1))
            |> List.unzip
        let env = List.fold (fun acc e -> Map.fold (fun acc' k v -> Map.add k v acc') acc e) Map.empty envs
        let sub = List.fold compose empty subs
        (env, sub)

    | ConsPat (hPat, tPat, _) ->
        // h :: t where h : 'a, t : 'a list
        let tv = fresh()
        let s1 = unify ty (TList tv)
        let (hEnv, s2) = inferPattern ctorEnv hPat (apply s1 tv)
        let (tEnv, s3) = inferPattern ctorEnv tPat (apply (compose s2 s1) (TList tv))
        let env = Map.fold (fun acc k v -> Map.add k v acc) hEnv tEnv
        let sub = compose s3 (compose s2 s1)
        (env, sub)

    | EmptyListPat _ ->
        // [] : 'a list
        let tv = fresh()
        let s = unify ty (TList tv)
        (Map.empty, s)

    | ConstPat (c, _) ->
        // Constant matches specific type
        let constType = match c with IntConst _ -> TInt | BoolConst _ -> TBool
        let s = unify ty constType
        (Map.empty, s)

    | ConstructorPat (name, argPatOpt, _) ->  // NEW
        // Look up constructor in environment
        match Map.tryFind name ctorEnv with
        | None ->
            raise (TypeException { Kind = UnboundConstructor name; Span = patternSpanOf pat })
        | Some ctorInfo ->
            // Instantiate constructor type with fresh type variables
            let freshVars = ctorInfo.TypeParams |> List.map (fun _ -> fresh())
            let subst = List.zip ctorInfo.TypeParams freshVars |> Map.ofList
            let resultType = apply subst ctorInfo.ResultType

            // Unify pattern type with constructor result type
            let s1 = unify ty resultType

            // Check argument pattern if present
            match (ctorInfo.ArgType, argPatOpt) with
            | (None, None) ->
                // Nullary constructor (e.g., None)
                (Map.empty, s1)
            | (Some argType, Some argPat) ->
                // Constructor with argument (e.g., Some x)
                let argType' = apply (compose s1 subst) argType
                let (env, s2) = inferPattern ctorEnv argPat argType'
                (env, compose s2 s1)
            | (None, Some _) ->
                raise (TypeException { Kind = ArityMismatch (name, 0, 1); Span = patternSpanOf pat })
            | (Some _, None) ->
                raise (TypeException { Kind = ArityMismatch (name, 1, 0); Span = patternSpanOf pat })
```

```fsharp
let rec synth (ctorEnv: ConstructorEnv) (env: TypeEnv) (expr: Expr) : Type * Subst =
    match expr with
    | Match (scrut, clauses, _) ->
        let (scrutType, s1) = synth ctorEnv env scrut
        // For each clause, infer pattern against scrutType
        let (clauseTypes, subs) =
            clauses
            |> List.map (fun (pat, body) ->
                let (patEnv, s2) = inferPattern ctorEnv pat (apply s1 scrutType)
                let env' = Map.fold (fun acc k (Scheme(vars, ty)) -> Map.add k (Scheme(vars, ty)) acc) (applyEnv (compose s2 s1) env) patEnv
                let (bodyType, s3) = synth ctorEnv env' body
                (bodyType, compose s3 (compose s2 s1)))
            |> List.unzip
        // Unify all clause body types
        let resultType = List.head clauseTypes
        let subs' = List.tail clauseTypes |> List.map (fun t -> unify resultType t) |> List.fold compose (List.last subs)
        (apply subs' resultType, subs')
    | ... // other cases, thread ctorEnv through recursive calls
```

```fsharp
let typeCheckModule (Module(decls, _): Module) : Result<unit, Diagnostic> =
    try
        // Separate type declarations from let declarations
        let typeDecls = decls |> List.choose (function Ast.TypeDecl td -> Some td | _ -> None)
        let letDecls = decls |> List.choose (function Ast.LetDecl(n, e, s) -> Some(n, e, s) | _ -> None)

        // Build constructor environment from type declarations
        let ctorEnv =
            typeDecls
            |> List.map Elaborate.elaborateTypeDecl
            |> List.fold (fun acc map -> Map.fold (fun acc' k v -> Map.add k v acc') acc map) Map.empty

        // Type check let declarations with constructor environment
        let env = Prelude.initialEnv  // Start with prelude bindings
        letDecls
        |> List.fold (fun (env: TypeEnv) (name, body, span) ->
            let (ty, _) = Bidir.synth ctorEnv env body
            let scheme = generalize env ty
            Map.add name scheme env) env
        |> ignore

        Ok ()
    with
    | TypeException diag -> Error diag
```
