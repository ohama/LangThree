Add new function to elaborate TypeDecl and build ConstructorEnv:
```fsharp
/// Elaborate a type declaration into ConstructorEnv entries
/// Example: type Option 'a = None | Some of 'a
/// Returns: Map with "None" -> {TypeParams=[0]; ArgType=None; ResultType=TData("Option",[TVar 0])}
///                  "Some" -> {TypeParams=[0]; ArgType=Some(TVar 0); ResultType=TData("Option",[TVar 0])}
let elaborateTypeDecl (TypeDecl(name, typeParams, constructors, _): Ast.TypeDecl) : ConstructorEnv =
    // Map type parameter names to TVar indices
    let paramMap =
        typeParams
        |> List.mapi (fun i p ->
            // Extract 'a -> 0, 'b -> 1
            let varName = if p.StartsWith("'") then p else "'" + p
            (varName, i))
        |> Map.ofList

    let typeParamVars = typeParams |> List.mapi (fun i _ -> i)
    let resultType = TData(name, List.map TVar typeParamVars)

    // Elaborate each constructor
    constructors
    |> List.map (fun (Ast.Constructor(ctorName, dataTypeOpt, _)) ->
        let argType =
            dataTypeOpt
            |> Option.map (fun te ->
                // Substitute type params with indices
                let rec subst = function
                    | TEVar v ->
                        match Map.tryFind v paramMap with
                        | Some idx -> TVar idx
                        | None -> failwithf "Unbound type parameter: %s" v
                    | TEInt -> TInt
                    | TEBool -> TBool
                    | TEString -> TString
                    | TEList te -> TList (subst te)
                    | TEArrow (t1, t2) -> TArrow (subst t1, subst t2)
                    | TETuple ts -> TTuple (List.map subst ts)
                subst te)

        let info = {
            TypeParams = typeParamVars
            ArgType = argType
            ResultType = resultType
        }
        (ctorName, info))
    |> Map.ofList
```

let testElaborateSimpleADT() =
    // type Bool = True | False
    let decl = Ast.TypeDecl("Bool", [], [
        Ast.Constructor("True", None, unknownSpan)
        Ast.Constructor("False", None, unknownSpan)
    ], unknownSpan)

    let ctorEnv = Elaborate.elaborateTypeDecl decl

    // Check True constructor
    match Map.tryFind "True" ctorEnv with
    | Some { TypeParams = []; ArgType = None; ResultType = TData("Bool", []) } -> ()
    | _ -> failwith "True constructor type mismatch"

    // Check False constructor
    match Map.tryFind "False" ctorEnv with
    | Some { TypeParams = []; ArgType = None; ResultType = TData("Bool", []) } -> ()
    | _ -> failwith "False constructor type mismatch"

[<Fact>]
let testElaborateParametricADT() =
