DECISION: Resolve type name from field names against RecordEnv in the evaluator. This is the simplest approach -- no AST changes needed, and field names are globally unique (enforced by the type checker in Plan 03).

In Eval.fs:
1. Add `recEnv: RecordEnv` parameter to the `eval` function signature (and `evalModule` / `evalDecl` if they exist as separate functions). Thread it through all recursive calls, similar to how ctorEnv is threaded.

2. Add a helper function to resolve type name from field names:
   ```fsharp
   let resolveRecordTypeName (recEnv: RecordEnv) (fieldNames: Set<string>) : string =
       recEnv
       |> Map.tryPick (fun typeName info ->
           let declFields = info.Fields |> List.map (fun f -> f.Name) |> Set.ofList
           if fieldNames = declFields then Some typeName else None)
       |> Option.defaultValue ""
   ```

3. Add `RecordValue` formatting in `formatValue`:
   ```fsharp
   | RecordValue (typeName, fields) ->
       let fieldStrs =
           fields
           |> Map.toList
           |> List.map (fun (name, value) -> sprintf "%s = %s" name (formatValue value))
       sprintf "{ %s }" (String.concat "; " fieldStrs)
   ```

4. Add `RecordPat` case in `matchPattern`:
   ```fsharp
   | RecordPat (fieldPats, _), RecordValue (_, fields) ->
       let bindings =
           fieldPats
           |> List.map (fun (fieldName, pat) ->
               match Map.tryFind fieldName fields with
               | Some value -> matchPattern pat value
               | None -> None)
       if List.forall Option.isSome bindings then
           Some (List.collect Option.get bindings)
       else
           None
   ```

5. Add `RecordExpr` evaluation in `eval` -- resolve type name from RecordEnv:
   ```fsharp
   | RecordExpr (_, fieldExprs, _) ->
       let fieldValues =
           fieldExprs
           |> List.map (fun (name, expr) -> (name, eval recEnv env expr))
           |> Map.ofList
       let fieldNames = fieldExprs |> List.map fst |> Set.ofList
       let typeName = resolveRecordTypeName recEnv fieldNames
       RecordValue (typeName, fieldValues)
   ```

6. Add `FieldAccess` evaluation:
   ```fsharp
   | FieldAccess (expr, fieldName, _) ->
       match eval recEnv env expr with
       | RecordValue (_, fields) ->
           match Map.tryFind fieldName fields with
           | Some value -> value
           | None -> failwithf "Field not found: %s" fieldName
       | v -> failwithf "Field access on non-record value: %s" (formatValue v)
   ```

7. Add `RecordUpdate` evaluation:
   ```fsharp
   | RecordUpdate (source, updates, _) ->
       match eval recEnv env source with
       | RecordValue (typeName, fields) ->
           let updatedFields =
               updates
               |> List.fold (fun acc (name, expr) ->
                   Map.add name (eval recEnv env expr) acc) fields
           RecordValue (typeName, updatedFields)
       | v -> failwithf "Copy-and-update on non-record value: %s" (formatValue v)
   ```

8. Add structural equality for RecordValue in `Equal` and `NotEqual`:
   ```fsharp
   // In Equal case:
   | RecordValue (t1, f1), RecordValue (t2, f2) -> BoolValue (t1 = t2 && f1 = f2)
   // In NotEqual case:
   | RecordValue (t1, f1), RecordValue (t2, f2) -> BoolValue (t1 <> t2 || f1 <> f2)
   ```
   NOTE: With proper type names resolved from RecordEnv, `t1 = t2` correctly distinguishes different record types that happen to have similar field values.
