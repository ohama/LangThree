IMPORTANT CONTEXT about current eval architecture:
- `eval` signature is: `eval (env: Env) (expr: Expr) : Value` -- NO ctorEnv parameter exists
- `evalMatchClauses` signature is: `evalMatchClauses (env: Env) (scrutinee: Value) (clauses: MatchClause list) : Value`
- `evalExpr` is a top-level convenience: `evalExpr (expr: Expr) : Value`
- There is NO `evalModule` function -- it does not exist
- There is NO `ctorEnv` parameter on eval -- constructors are handled differently (as DataValue)

1. Add `recEnv: RecordEnv` as FIRST parameter to `eval`:
   Change from: `and eval (env: Env) (expr: Expr) : Value`
   To: `and eval (recEnv: RecordEnv) (env: Env) (expr: Expr) : Value`

2. Add `recEnv: RecordEnv` as FIRST parameter to `evalMatchClauses`:
   Change from: `and evalMatchClauses (env: Env) (scrutinee: Value) (clauses: MatchClause list) : Value`
   To: `and evalMatchClauses (recEnv: RecordEnv) (env: Env) (scrutinee: Value) (clauses: MatchClause list) : Value`

3. Update ALL recursive calls to `eval` within Eval.fs to pass `recEnv` through:
   - Every `eval env ...` becomes `eval recEnv env ...`
   - Every `evalMatchClauses env ...` becomes `evalMatchClauses recEnv env ...`
   This is a mechanical find-and-replace within the file. There are approximately 15-20 call sites based on grep results.

4. Update `evalExpr` to pass `Map.empty` as recEnv:
   ```fsharp
   let evalExpr (expr: Expr) : Value =
       eval Map.empty emptyEnv expr
   ```

5. Add a helper function to resolve type name from field names:
   ```fsharp
   let resolveRecordTypeName (recEnv: RecordEnv) (fieldNames: Set<string>) : string =
       recEnv
       |> Map.tryPick (fun typeName info ->
           let declFields = info.Fields |> List.map (fun f -> f.Name) |> Set.ofList
           if fieldNames = declFields then Some typeName else None)
       |> Option.defaultValue ""
   ```

6. Add `RecordValue` formatting in `formatValue`:
   ```fsharp
   | RecordValue (typeName, fields) ->
       let fieldStrs =
           fields
           |> Map.toList
           |> List.map (fun (name, value) -> sprintf "%s = %s" name (formatValue value))
       sprintf "{ %s }" (String.concat "; " fieldStrs)
   ```

7. Add `RecordPat` case in `matchPattern`:
   ```fsharp
   | RecordPat (fieldPats, _), RecordValue (_, fields) ->
       let bindings =
           fieldPats
           |> List.map (fun (fieldName, pat) ->
               match Map.tryFind fieldName fields with
               | Some value -> matchPattern pat value
               | None -> None)
       if List.forall Option.isSome bindings then
           Some (List.collect Option.get bindings)
       else
           None
   ```

8. Add `RecordExpr` evaluation in `eval` -- resolve type name from RecordEnv:
   ```fsharp
   | RecordExpr (_, fieldExprs, _) ->
       let fieldValues =
           fieldExprs
           |> List.map (fun (name, expr) -> (name, eval recEnv env expr))
           |> Map.ofList
       let fieldNames = fieldExprs |> List.map fst |> Set.ofList
       let typeName = resolveRecordTypeName recEnv fieldNames
       RecordValue (typeName, fieldValues)
   ```

9. Add `FieldAccess` evaluation:
   ```fsharp
   | FieldAccess (expr, fieldName, _) ->
       match eval recEnv env expr with
       | RecordValue (_, fields) ->
           match Map.tryFind fieldName fields with
           | Some value -> value
           | None -> failwithf "Field not found: %s" fieldName
       | v -> failwithf "Field access on non-record value: %s" (formatValue v)
   ```

10. Add `RecordUpdate` evaluation:
    ```fsharp
    | RecordUpdate (source, updates, _) ->
        match eval recEnv env source with
        | RecordValue (typeName, fields) ->
            let updatedFields =
                updates
                |> List.fold (fun acc (name, expr) ->
                    Map.add name (eval recEnv env expr) acc) fields
            RecordValue (typeName, updatedFields)
        | v -> failwithf "Copy-and-update on non-record value: %s" (formatValue v)
    ```

11. Add structural equality for RecordValue in `Equal` and `NotEqual`:
    ```fsharp
    // In Equal case:
    | RecordValue (t1, f1), RecordValue (t2, f2) -> BoolValue (t1 = t2 && f1 = f2)
    // In NotEqual case:
    | RecordValue (t1, f1), RecordValue (t2, f2) -> BoolValue (t1 <> t2 || f1 <> f2)
    ```
